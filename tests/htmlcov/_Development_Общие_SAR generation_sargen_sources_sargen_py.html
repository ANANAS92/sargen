15def clamp_array(array: np.ndarray, minimum: Number = 0, maximum: Number = 255) -> np.ndarray: 

16 """ 

17 Clamp values less then 0 to 0 and greater then 255 to 255. 

18 

19 :param array: array of numbers. 

20 :param minimum: minimum value to clamp elements below it. 

21 :param maximum: maximum value to clamp elements above it. 

22 :return: arrays of same shape with clamped values. 

23 """ 

24 # noinspection PyTypeChecker 

25 return np.where(array < minimum, minimum, np.where(array > maximum, maximum, array))

28def convert_image(image: np.ndarray, *, channels_weights: np.ndarray = np.array([0.299, 0.557, 0.144]), bias: float = -33) -> np.ndarray: 

29 """ 

30 Convert R, G, B color channels to luma single channel. 

31 

32 Note that luma is not clamped. 

33 

34 :param image: array of pixels of shape (height, width, 3) of floats. 

35 :param channels_weights: R, G, B coefficients array of shape 3 of floats. 

36 :param bias: luma bias to apply after conversion. 

37 :return: array of shape (height, width) of floats. 

38 """ 

39 return (image * channels_weights).sum(axis=2) + bias

57def rotate_image(image: Image.Image, *, generator: np.random.Generator = np.random.default_rng()) -> Image.Image: 

58 """ 

59 Rotate image by uniformly random angle. 

60 

61 Note that image is cropped and converted to RGB (same channels' values in grayscale case). 

62 

63 :param image: image to rotate and crop. 

64 :param generator: random number generator. 

65 """ 

66 # Rotation over 360 degrees crops image by circle with radius of min(width, height). So we need to crop insquare of that circle. 

67 angle = generator.uniform(0, 360) 

68 center = np.array(image.size) / 2 

69 insquare_side = min(*image.size) / math.sqrt(2) 

70 insquare = np.concatenate([center - insquare_side / 2, center + insquare_side / 2]) 

71 image = image.rotate(angle, resample=Image.BICUBIC, expand=False, center=tuple(center)).crop(tuple(insquare)) 

72 image.load() # cropping is lazy so we need to load results 

73 return image

91def _find_perspective_coefficients(origin_points: np.ndarray, transformed_points: np.ndarray) -> np.ndarray: 

92 """ 

93 Find perspective transformation coefficients a, b, c, d, e, f, g, h. 

94 

95 See https://stackoverflow.com/a/14178717 along with other answers and comments. 

96 

97 :param origin_points: 4 origin points that are mapped to transformed (as array of shape (4, 2) of floats). 

98 :param transformed_points: 4 points to where origin ones are mapped (as array of shape (4, 2) of floats). 

99 :return: array of shape 8 of floats. 

100 """ 

101 matrix_buf = [] 

102 for src_point, dist_point in zip(origin_points, transformed_points): 

103 matrix_buf.append([src_point[0], src_point[1], 1, 0, 0, 0, -dist_point[0] * src_point[0], -dist_point[0] * src_point[1]]) 

104 matrix_buf.append([0, 0, 0, src_point[0], src_point[1], 1, -dist_point[1] * src_point[0], -dist_point[1] * src_point[1]]) 

105 

106 a_matrix = np.array(matrix_buf, dtype=np.float) 

107 b_matrix = np.array(transformed_points).reshape(8) 

108 

109 res = np.linalg.solve(a_matrix, b_matrix) 

110 return res.reshape(8)

113def tilt_image(image: Image.Image, *, deviation: float = 0.2, generator: np.random.Generator = np.random.default_rng()) -> Image.Image: 

114 """ 

115 Tilt image to any direction by gaussian random factor. This is perspective transformation. 

116 

117 Note that image is cropped to fit origin box. 

118 

119 :param image: image to tilt and crop. 

120 :param deviation: maximum perspective in pixels as the smallest side multiplication factor, e.g. 0.2 from 1920x1080 results in deviation from 0 to 216 pixels in all directions. 

121 :param generator: random number generator. 

122 """ 

123 # See https://stackoverflow.com/a/14178717 along with other answers and comments. 

124 # We have 4 points = 8 coordinates that can be shifted (each in one direction only). 

125 factors = -1 * clamp_array(generator.uniform(0, deviation * min(image.size), 8), minimum=0, maximum=deviation * min(image.size)) # invert factors to stretch image out of the clipping box 

126 # Shift image box points out of self by factors to avoid empty space after perspective. 

127 origin_points = np.array([[0, 0], [image.width, 0], [image.width, image.height], [0, image.height]]) 

128 transformed_points = np.array([[-factors[0], -factors[1]], [image.width + factors[2], -factors[3]], [image.width + factors[4], image.height + factors[5]], [-factors[6], image.height + factors[7]]]) 

129 

130 transformation_parameters = _find_perspective_coefficients(origin_points, transformed_points) 

131 return image.transform(image.size, Image.PERSPECTIVE, tuple(transformation_parameters), Image.BICUBIC)

134def transform_image(image: Image.Image, *, rotate: bool = True, scale: bool = True, tilt: bool = True, scale_mean: float = 1, scale_deviation: float = 0.2, tilt_deviation: float = 0.2, generator: np.random.Generator = np.random.default_rng()) -> Image.Image: 

135 """ 

136 Apply random transformations to the image: rotation (by uniformly random angle), scale (by gaussian random factor) and tilt (by gaussian random factor). 

137 

138 Note that image is cropped and converted to RGB (same channels' values in grayscale case). 

139 

140 See also rotate_image, scale_image, tilt_image. 

141 

142 :param image: image to apply transformations to. 

143 :param rotate: whether to apply rotation transformation. 

144 :param scale: whether to apply rotation transformation. 

145 :param tilt: whether to apply rotation transformation. 

146 :param scale_mean: mu parameter of normal distribution in case scale is True. 

147 :param scale_deviation: sigma parameter of normal distribution also known as standard deviation in case scale is True. 

148 :param tilt_deviation: maximum perspective in pixels as the smallest side multiplication factor in case tilt is True, e.g. 0.2 from 1920x1080 results in deviation from 0 to 216 pixels in all directions. 

149 :param generator: random number generator. 

150 """ 

151 if rotate: 

152 image = rotate_image(image, generator=generator) 

153 if scale: 

154 image = scale_image(image, mean=scale_mean, deviation=scale_deviation, generator=generator) 

155 if tilt: 

156 image = tilt_image(image, deviation=tilt_deviation, generator=generator) 

157 return image

160def sar(image: Union[str, Path, Image.Image, np.ndarray], *, noise: bool = True, transform: bool = True, channels_weights: np.ndarray = np.array([0.299, 0.557, 0.144]), bias: float = -33, noise_mean: float = 0, noise_deviation: float = 12.5, scale_mean: float = 1, scale_deviation: float = 0.2, tilt_deviation: float = 0.2, generator: np.random.Generator = np.random.default_rng()) -> Image.Image: 

161 """ 

162 Generate SAR image from optic one. 

163 

164 Source image must contain 3 first channels: R, G, B. Resulting image contains also 3 channels: grayscale, grayscale, grayscale due to compatibility with popular image formats and processing libraries or programs. 

165 

166 Generation parameters can be tuned with keyword arguments. 

167 

168 :param image: text of path to image file or path object or image object or array of shape (height, width, 3) of numbers. 

169 :param noise: whether to apply gaussian noise. 

170 :param transform: whether to apply random transformations. 

171 :param channels_weights: R, G, B coefficients array of shape 3 of floats. 

172 :param bias: luma bias to apply after conversion. 

173 :param noise_mean: mu parameter of normal distribution in case noise is True. 

174 :param noise_deviation: sigma parameter of normal distribution also known as standard deviation in case noise is True. 

175 :param scale_mean: mu parameter of normal distribution in case scale is True. 

176 :param scale_deviation: sigma parameter of normal distribution also known as standard deviation in case scale is True. 

177 :param tilt_deviation: maximum perspective in pixels as the smallest side multiplication factor in case tilt is True, e.g. 0.2 from 1920x1080 results in deviation from 0 to 216 pixels in all directions. 

178 :param generator: random number generator. 

179 """ 

180 if isinstance(image, str): 

181 image = Path(image) 

182 if isinstance(image, Path): 

183 if not image.is_file(): 

184 raise ValueError(f'Path {image.absolute()} does not point to an image file.') 

185 image = Image.open(image) 

186 if isinstance(image, Image.Image): 

187 # noinspection PyTypeChecker 

188 image = np.array(image) 

189 shape = image.shape 

190 if len(shape) < 3 or shape[2] < 3: 

191 raise ValueError(f'Image must has shape (height, width, 3) but has {shape}.') 

192 image = image[:, :, :3] 

193 if not isinstance(image, np.ndarray): 

194 raise ValueError('Bad data type for image.') 

195 image = image.astype(float) 

196 image = convert_image(image, channels_weights=channels_weights, bias=bias) 

197 if noise: 

198 image = noise_image(image, mean=noise_mean, deviation=noise_deviation, generator=generator) 

199 image = Image.fromarray(clamp_array(image, minimum=0, maximum=255).astype(np.uint8)).convert('RGB') 

200 if transform: 

201 image = transform_image(image, scale_mean=scale_mean, scale_deviation=scale_deviation, tilt_deviation=tilt_deviation, generator=generator)
